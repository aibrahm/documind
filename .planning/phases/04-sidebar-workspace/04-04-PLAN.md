```typescript
// src/app/(workspace)/projects/[slug]/_tabs/documents.tsx
"use client";
import { useEffect, useState } from "react";
import Link from "next/link";
import { FileText, ExternalLink } from "lucide-react";
import { Tag } from "@/components/ui-system";

interface LinkedDocument {
  id: string;
  title: string;
  type: string | null;
  classification: string | null;
  language: string | null;
  page_count: number | null;
  created_at: string | null;
  link: { role: string | null; added_by: string | null; added_at: string };
}

export function DocumentsTab({ projectSlug }: { projectSlug: string }) {
  const [docs, setDocs] = useState<LinkedDocument[] | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;
    (async () => {
      try {
        const res = await fetch(`/api/projects/${projectSlug}/documents`);
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        const data = await res.json();
        if (!cancelled) setDocs(data.documents || []);
      } catch (err) {
        if (!cancelled) setError(err instanceof Error ? err.message : "Failed to load documents");
      }
    })();
    return () => {
      cancelled = true;
    };
  }, [projectSlug]);

  if (error) return <ErrorState message={error} />;
  if (docs === null) return <LoadingState label="Loading documents" />;
  if (docs.length === 0) {
    return (
      <EmptyState
        title="No documents linked yet"
        body="Upload documents from the Upload page, then link them to this project."
      />
    );
  }

  return (
    <div className="h-full overflow-y-auto">
      <div className="max-w-3xl mx-auto px-6 py-8 space-y-2">
        {docs.map((d) => (
          <Link
            key={d.id}
            href={`/documents/${d.id}`}
            className="flex items-start gap-3 rounded-lg border border-border px-4 py-3 hover:bg-surface-raised transition-colors"
          >
            <FileText className="w-4 h-4 text-ink-muted mt-0.5 flex-shrink-0" />
            <div className="flex-1 min-w-0">
              <div className="flex items-start justify-between gap-2">
                <h3 className="text-sm font-medium text-ink truncate">{d.title}</h3>
                <ExternalLink className="w-3.5 h-3.5 text-ink-faint flex-shrink-0" />
              </div>
              <div className="flex items-center gap-2 mt-1 text-xs text-ink-muted">
                {d.type && <Tag variant="default">{d.type}</Tag>}
                {d.classification && <Tag variant="default">{d.classification}</Tag>}
                {d.page_count != null && <span>{d.page_count} pages</span>}
                {d.link.role && <span>· role: {d.link.role}</span>}
              </div>
            </div>
          </Link>
        ))}
      </div>
    </div>
  );
}

function LoadingState({ label }: { label: string }) {
  return (
    <div className="h-full flex items-center justify-center text-sm text-ink-muted">
      {label}…
    </div>
  );
}

function EmptyState({ title, body }: { title: string; body: string }) {
  return (
    <div className="h-full flex items-center justify-center">
      <div className="max-w-md text-center space-y-2">
        <p className="dm-heading text-ink">{title}</p>
        <p className="text-sm text-ink-muted">{body}</p>
      </div>
    </div>
  );
}

function ErrorState({ message }: { message: string }) {
  return (
    <div className="h-full flex items-center justify-center">
      <div className="max-w-md rounded border border-danger bg-danger-bg text-danger px-4 py-3 text-sm">
        Failed to load documents: {message}
      </div>
    </div>
  );
}
```

// src/app/(workspace)/projects/[slug]/_tabs/negotiations.tsx
"use client";
import { useEffect, useState } from "react";
import { Handshake } from "lucide-react";
import { Tag } from "@/components/ui-system";

interface Negotiation {
  id: string;
  name: string;
  status: string;
  key_terms: Record<string, unknown> | null;
  opened_at: string | null;
  closed_at: string | null;
}

const STATUS_VARIANT: Record<string, "default" | "blue" | "green" | "amber" | "red"> = {
  open: "blue",
  active: "green",
  stalled: "amber",
  closed_won: "green",
  closed_lost: "red",
  withdrawn: "default",
};

export function NegotiationsTab({ projectId }: { projectId: string }) {
  const [negs, setNegs] = useState<Negotiation[] | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;
    (async () => {
      try {
        const res = await fetch(`/api/negotiations?project_id=${projectId}`);
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        const data = await res.json();
        if (!cancelled) setNegs(data.negotiations || []);
      } catch (err) {
        if (!cancelled) setError(err instanceof Error ? err.message : "Failed to load negotiations");
      }
    })();
    return () => {
      cancelled = true;
    };
  }, [projectId]);

  if (error) {
    return (
      <div className="h-full flex items-center justify-center">
        <div className="max-w-md rounded border border-danger bg-danger-bg text-danger px-4 py-3 text-sm">
          Failed to load negotiations: {error}
        </div>
      </div>
    );
  }
  if (negs === null) {
    return <div className="h-full flex items-center justify-center text-sm text-ink-muted">Loading negotiations…</div>;
  }
  if (negs.length === 0) {
    return (
      <div className="h-full flex items-center justify-center">
        <div className="max-w-md text-center space-y-2">
          <Handshake className="w-8 h-8 text-ink-faint mx-auto" />
          <p className="dm-heading text-ink">No negotiations yet</p>
          <p className="text-sm text-ink-muted">
            Create a negotiation thread to track a specific deal scenario (e.g., &ldquo;Scenario 1 — Developer + Partnership&rdquo;).
            The deal-room UI lands in Phase 06.
          </p>
        </div>
      </div>
    );
  }

  return (
    <div className="h-full overflow-y-auto">
      <div className="max-w-3xl mx-auto px-6 py-8 space-y-3">
        {negs.map((n) => (
          <div key={n.id} className="rounded-lg border border-border px-4 py-3">
            <div className="flex items-start justify-between gap-2">
              <h3 className="text-sm font-medium text-ink">{n.name}</h3>
              <Tag variant={STATUS_VARIANT[n.status] || "default"}>{n.status.replace("_", " ")}</Tag>
            </div>
            {n.key_terms && Object.keys(n.key_terms).length > 0 && (
              <div className="mt-2 grid grid-cols-2 gap-x-4 gap-y-1 text-xs text-ink-muted">
                {Object.entries(n.key_terms).slice(0, 6).map(([k, v]) => (
                  <div key={k} className="flex items-center gap-1.5">
                    <span className="text-ink-faint">{k.replace(/_/g, " ")}:</span>
                    <span className="text-ink">{String(v)}</span>
                  </div>
                ))}
              </div>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}
```

"use client";
import { useEffect, useState } from "react";
import Link from "next/link";
import { MessageSquare } from "lucide-react";

interface ConversationItem {
  id: string;
  title: string;
  mode: string | null;
  query: string | null;
  created_at: string | null;
}

export function ChatsTab({ projectSlug }: { projectSlug: string }) {
  const [convos, setConvos] = useState<ConversationItem[] | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;
    (async () => {
      try {
        const res = await fetch(`/api/projects/${projectSlug}/conversations`);
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        const data = await res.json();
        if (!cancelled) setConvos(data.conversations || []);
      } catch (err) {
        if (!cancelled) setError(err instanceof Error ? err.message : "Failed to load conversations");
      }
    })();
    return () => {
      cancelled = true;
    };
  }, [projectSlug]);

  if (error) {
    return (
      <div className="h-full flex items-center justify-center">
        <div className="max-w-md rounded border border-danger bg-danger-bg text-danger px-4 py-3 text-sm">
          Failed to load conversations: {error}
        </div>
      </div>
    );
  }
  if (convos === null) {
    return <div className="h-full flex items-center justify-center text-sm text-ink-muted">Loading conversations…</div>;
  }
  if (convos.length === 0) {
    return (
      <div className="h-full flex items-center justify-center">
        <div className="max-w-md text-center space-y-2">
          <MessageSquare className="w-8 h-8 text-ink-faint mx-auto" />
          <p className="dm-heading text-ink">No conversations yet</p>
          <p className="text-sm text-ink-muted">
            Start chatting in the Overview tab — new conversations will appear here.
          </p>
        </div>
      </div>
    );
  }

  return (
    <div className="h-full overflow-y-auto">
      <div className="max-w-3xl mx-auto px-6 py-8 space-y-2">
        {convos.map((c) => (
          <Link
            key={c.id}
            href={`/?conversation=${c.id}`}
            className="block rounded-lg border border-border px-4 py-3 hover:bg-surface-raised transition-colors"
          >
            <div className="flex items-start justify-between gap-2">
              <h3 className="text-sm font-medium text-ink truncate">{c.title}</h3>
              {c.mode && <span className="text-xs text-ink-faint uppercase tracking-wide">{c.mode}</span>}
            </div>
            {c.query && (
              <p className="mt-1 text-xs text-ink-muted line-clamp-1">{c.query}</p>
            )}
            {c.created_at && (
              <p className="mt-1 text-xs text-ink-faint">{new Date(c.created_at).toLocaleString()}</p>
            )}
          </Link>
        ))}
      </div>
    </div>
  );
}
```
