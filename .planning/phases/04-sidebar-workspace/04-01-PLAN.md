```typescript
// src/app/(workspace)/layout.tsx
import { Nav } from "@/components/nav";

export default function WorkspaceLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <div className="h-screen flex flex-col">
      <Nav />
      <div className="flex-1 flex min-h-0 overflow-hidden">
        {children}
      </div>
    </div>
  );
}
```

**Why `flex-1 flex min-h-0 overflow-hidden`:** the existing pages render their own inner sidebar + content split. After migration, the layout provides the vertical flex (Nav on top, flex-1 row below) and each page fills that row. The `min-h-0` is critical so children can shrink below their content height inside a flex parent.
