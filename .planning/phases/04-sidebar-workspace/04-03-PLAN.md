// src/app/(workspace)/projects/[slug]/page.tsx
import { notFound } from "next/navigation";
import { supabaseAdmin } from "@/lib/supabase";
import { WorkspaceClient } from "./workspace-client";

export default async function ProjectWorkspacePage({
  params,
  searchParams,
}: {
  params: Promise<{ slug: string }>;
  searchParams: Promise<{ tab?: string }>;
}) {
  const { slug } = await params;
  const { tab = "overview" } = await searchParams;

  const { data: project, error } = await supabaseAdmin
    .from("projects")
    .select("*")
    .eq("slug", slug)
    .maybeSingle();

  if (error || !project) {
    notFound();
  }

  // Fetch membership counts + companies in parallel
  const [docsCount, companiesLinks, negsCount, convosCount] = await Promise.all([
    supabaseAdmin
      .from("project_documents")
      .select("project_id", { count: "exact", head: true })
      .eq("project_id", project.id),
    supabaseAdmin
      .from("project_companies")
      .select(`
        role,
        entity:entities ( id, name, name_en, type )
      `)
      .eq("project_id", project.id),
    supabaseAdmin
      .from("negotiations")
      .select("id", { count: "exact", head: true })
      .eq("project_id", project.id),
    supabaseAdmin
      .from("conversations")
      .select("id", { count: "exact", head: true })
      .eq("project_id", project.id),
  ]);

  const counterparties = (companiesLinks.data || [])
    .filter((l) => l.entity)
    .map((l) => ({
      id: (l.entity as { id: string }).id,
      name: (l.entity as { name: string }).name,
      name_en: (l.entity as { name_en: string | null }).name_en,
      role: l.role as string,
    }));

  const counts = {
    documents: docsCount.count || 0,
    companies: counterparties.length,
    negotiations: negsCount.count || 0,
    conversations: convosCount.count || 0,
  };

  return (
    <WorkspaceClient
      project={project}
      initialTab={tab}
      counts={counts}
      counterparties={counterparties}
    />
  );
}
```

// src/components/project-tabs.tsx
"use client";
import { useRouter, useSearchParams, usePathname } from "next/navigation";

const TABS = [
  { id: "overview", label: "Overview" },
  { id: "documents", label: "Documents" },
  { id: "negotiations", label: "Negotiations" },
  { id: "chats", label: "Chats" },
  { id: "memory", label: "Memory" },
] as const;

export type TabId = (typeof TABS)[number]["id"];

interface ProjectTabsProps {
  activeTab: TabId;
}

export function ProjectTabs({ activeTab }: ProjectTabsProps) {
  const router = useRouter();
  const pathname = usePathname();
  const searchParams = useSearchParams();

  const handleClick = (tab: TabId) => {
    const sp = new URLSearchParams(searchParams.toString());
    if (tab === "overview") {
      sp.delete("tab");
    } else {
      sp.set("tab", tab);
    }
    const query = sp.toString();
    router.push(query ? `${pathname}?${query}` : pathname, { scroll: false });
  };

  return (
    <nav className="border-b border-border px-6 flex gap-1" role="tablist">
      {TABS.map((tab) => (
        <button
          key={tab.id}
          role="tab"
          aria-selected={activeTab === tab.id}
          onClick={() => handleClick(tab.id)}
          className={`px-4 py-2.5 text-sm font-medium border-b-2 -mb-px transition-colors ${
            activeTab === tab.id
              ? "border-ink text-ink"
              : "border-transparent text-ink-muted hover:text-ink"
          }`}
        >
          {tab.label}
        </button>
      ))}
    </nav>
  );
}
```

"use client";
import { useState, useEffect } from "react";
import { useSearchParams } from "next/navigation";
import type { Database } from "@/lib/database.types";
import { useChat } from "@/lib/hooks/use-chat";
import { ProjectWorkspaceHeader } from "@/components/project-workspace-header";
import { ProjectTabs, type TabId } from "@/components/project-tabs";
import { OverviewTab } from "./_tabs/overview";

type Project = Database["public"]["Tables"]["projects"]["Row"];

interface Counterparty {
  id: string;
  name: string;
  name_en: string | null;
  role: string;
}

interface WorkspaceClientProps {
  project: Project;
  initialTab: string;
  counts: {
    documents: number;
    companies: number;
    negotiations: number;
    conversations: number;
  };
  counterparties: Counterparty[];
}

const VALID_TABS: TabId[] = ["overview", "documents", "negotiations", "chats", "memory"];

function isValidTab(t: string): t is TabId {
  return (VALID_TABS as string[]).includes(t);
}

export function WorkspaceClient({
  project,
  initialTab,
  counts,
  counterparties,
}: WorkspaceClientProps) {
  const searchParams = useSearchParams();
  const rawTab = searchParams.get("tab") ?? initialTab;
  const activeTab: TabId = isValidTab(rawTab) ? rawTab : "overview";

  // Chat state lives here so it survives tab switches
  const chat = useChat({ projectId: project.id });

  return (
    <div className="flex-1 flex flex-col min-w-0 overflow-hidden">
      <ProjectWorkspaceHeader
        project={project}
        counterparties={counterparties}
        counts={counts}
      />
      <ProjectTabs activeTab={activeTab} />

      {/* Tab content — using `hidden` CSS so tabs stay mounted and preserve state */}
      <div className="flex-1 min-h-0 overflow-hidden">
        <div hidden={activeTab !== "overview"} className="h-full">
          <OverviewTab project={project} counts={counts} chat={chat} />
        </div>
        <div hidden={activeTab !== "documents"} className="h-full">
          <TabPlaceholder label="Documents" />
        </div>
        <div hidden={activeTab !== "negotiations"} className="h-full">
          <TabPlaceholder label="Negotiations" />
        </div>
        <div hidden={activeTab !== "chats"} className="h-full">
          <TabPlaceholder label="Chats" />
        </div>
        <div hidden={activeTab !== "memory"} className="h-full">
          <TabPlaceholder label="Memory" />
        </div>
      </div>
    </div>
  );
}

function TabPlaceholder({ label }: { label: string }) {
  return (
    <div className="h-full flex items-center justify-center">
      <div className="text-center space-y-2">
        <p className="dm-heading text-ink">{label} tab</p>
        <p className="text-sm text-ink-muted">
          Coming in Phase 04-04 — this tab exists so the tab bar is functional.
        </p>
      </div>
    </div>
  );
}

// src/app/(workspace)/projects/[slug]/_tabs/overview.tsx
"use client";
import { useRef } from "react";
import type { Database } from "@/lib/database.types";
import type { UseChatResult } from "@/lib/hooks/use-chat";
import { ChatInput, type ChatInputHandle } from "@/components/chat-input";
import { ChatMessage } from "@/components/chat-message";
import { FileText, Building2, Handshake, MessageSquare } from "lucide-react";

type Project = Database["public"]["Tables"]["projects"]["Row"];

interface OverviewTabProps {
  project: Project;
  counts: {
    documents: number;
    companies: number;
    negotiations: number;
    conversations: number;
  };
  chat: UseChatResult;
}

export function OverviewTab({ project, counts, chat }: OverviewTabProps) {
  const chatInputRef = useRef<ChatInputHandle>(null);
  const { messages, streaming, streamingContent, routingStatus, error, send } = chat;

  const isIdle = messages.length === 0 && !streaming;

  return (
    <div className="h-full flex flex-col overflow-hidden">
      {/* Scrollable content */}
      <div className="flex-1 overflow-y-auto">
        <div className="max-w-3xl mx-auto px-6 py-8 space-y-6">
          {isIdle ? (
            <IdleState project={project} counts={counts} />
          ) : (
            <MessageThread
              messages={messages}
              streaming={streaming}
              streamingContent={streamingContent}
              routingStatus={routingStatus}
              error={error}
            />
          )}
        </div>
      </div>

      {/* Fixed chat input at the bottom */}
      <div className="border-t border-border bg-surface">
        <div className="max-w-3xl mx-auto px-6 py-4">
          <ChatInput
            ref={chatInputRef}
            onSend={send}
            disabled={streaming}
            placeholder={`Ask something about ${project.name}… (type @ to mention)`}
          />
        </div>
      </div>
    </div>
  );
}

function IdleState({
  project,
  counts,
}: {
  project: Project;
  counts: OverviewTabProps["counts"];
}) {
  return (
    <div className="space-y-8">
      {/* Greeting */}
      <div>
        <h2 className="dm-heading-xl text-ink">{project.name}</h2>
        {project.context_summary && (
          <p className="mt-2 text-ink-muted leading-relaxed">
            {project.context_summary}
          </p>
        )}
      </div>

      {/* Count cards */}
      <div className="grid grid-cols-4 gap-3">
        <CountCard icon={FileText} label="Documents" value={counts.documents} />
        <CountCard icon={Building2} label="Companies" value={counts.companies} />
        <CountCard icon={Handshake} label="Negotiations" value={counts.negotiations} />
        <CountCard icon={MessageSquare} label="Conversations" value={counts.conversations} />
      </div>

      {/* Starter prompt */}
      <div className="rounded-lg border border-border bg-surface-raised px-4 py-3 text-sm text-ink-muted">
        Start by asking anything — the model has access to this project&apos;s documents
        and context. For now, retrieval is global; project-scoped chat lands in Phase 05.
      </div>
    </div>
  );
}

function MessageThread({
  messages,
  streaming,
  streamingContent,
  routingStatus,
  error,
}: {
  messages: UseChatResult["messages"];
  streaming: boolean;
  streamingContent: string;
  routingStatus: string | null;
  error: string | null;
}) {
  return (
    <div className="space-y-6">
      {messages.map((msg, i) => (
        <ChatMessage
          key={i}
          role={msg.role}
          content={msg.content}
          metadata={msg.metadata}
        />
      ))}
      {streaming && (
        <ChatMessage
          role="assistant"
          content={streamingContent}
          isStreaming
          metadata={{ mode: routingStatus || undefined }}
        />
      )}
      {error && (
        <div className="rounded border border-danger bg-danger-bg text-danger px-3 py-2 text-sm">
          {error}
        </div>
      )}
    </div>
  );
}

function CountCard({
  icon: Icon,
  label,
  value,
}: {
  icon: React.ComponentType<{ className?: string }>;
  label: string;
  value: number;
}) {
  return (
    <div className="border border-border rounded-lg px-4 py-3 bg-surface">
      <div className="flex items-center gap-2 text-ink-muted text-xs">
        <Icon className="w-3.5 h-3.5" />
        {label}
      </div>
      <div className="mt-1 dm-heading text-ink">{value}</div>
    </div>
  );
}
